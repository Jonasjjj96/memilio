168 struct LogInfectionState : mio::LogAlways {
169  using Type = std::pair<mio::abm::TimePoint, Eigen::VectorXd>;
175  static Type log(const mio::abm::Simulation& sim)
176  {
177 
178  Eigen::VectorXd sum = Eigen::VectorXd::Zero(Eigen::Index(mio::abm::InfectionState::Count));
179  auto curr_time = sim.get_time();
180  PRAGMA_OMP(for)
181  for (auto& location : sim.get_model().get_locations()) {
182  for (uint32_t inf_state = 0; inf_state < (int)mio::abm::InfectionState::Count; inf_state++) {
183  sum[inf_state] += sim.get_model().get_subpopulation(location.get_id(), curr_time,
184  mio::abm::InfectionState(inf_state));
185  }
186  }
187  return std::make_pair(curr_time, sum);
188  }
189 };

215 template <class... Loggers>
216 struct DataWriterToMemoryDelta {
217  using Data = std::tuple<std::vector<typename Loggers::Type>...>;
218  template <class Logger>
225  static void add_record(const typename Logger::Type& t, Data& data)
226  {
227 
228  if (std::get<index_of_type_v<Logger, Loggers...>>(data).size() > 0) {
229  typename Logger::Type diff_vector{};
230  auto& current_state_vec = std::get<index_of_type_v<Logger, Loggers...>>(data).front();
231  for (auto i = 0; i < (int)current_state_vec.size(); i++) {
232  if (std::get<1>(t[i]) != std::get<1>(current_state_vec[i])) {
233  std::get<1>(current_state_vec[i]) = std::get<1>(t[i]);
234  diff_vector.push_back(t[i]);
235  }
236  }
237  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(diff_vector);
238  }
239  else {
240  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(
241  t); // We use the first entry as a reference for the current position.
242  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(t);
243  }
244  }
245 };
