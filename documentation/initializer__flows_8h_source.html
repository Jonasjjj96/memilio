85  bool compute_initialization_vector(ScalarType total_population, ScalarType deaths, ScalarType total_confirmed_cases)
86  {

88  Eigen::VectorXd init(LctState::Count);
89 
90  // Exposed.
91  init.segment(LctState::template get_first_index<InfectionState::Exposed>(),
92  LctState::template get_num_subcompartments<InfectionState::Exposed>()) =
93  compute_compartment<InfectionState::Exposed>((Eigen::Index)InfectionTransition::SusceptibleToExposed,
94  1 / m_model.parameters.template get<TimeExposed>());
95 
96  if ((init.segment(LctState::template get_first_index<InfectionState::Exposed>(),
97  LctState::template get_num_subcompartments<InfectionState::Exposed>())
98  .array() < 0)
99  .any()) {
100  // Something went wrong while calculating the initial values for the subcompartments.
101  // See the compute_compartment() function for details.
102  return true;
103  }
104  // InfectedNoSymptoms.
105  init.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
106  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>()) =
107  compute_compartment<InfectionState::InfectedNoSymptoms>(
108  (Eigen::Index)InfectionTransition::ExposedToInfectedNoSymptoms,
109  1 / m_model.parameters.template get<TimeInfectedNoSymptoms>());
110 
111  if ((init.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
112  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
113  .array() < 0)
114  .any()) {
115  // Something went wrong while calculating the initial values for the subcompartments.
116  // See the compute_compartment() function for details.
117  return true;
118  }
119  // InfectedSymptoms.

121  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>()) =
122  compute_compartment<InfectionState::InfectedSymptoms>(
123  (Eigen::Index)InfectionTransition::InfectedNoSymptomsToInfectedSymptoms,
124  1 / m_model.parameters.template get<TimeInfectedSymptoms>());
125 
126  if ((init.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
127  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
128  .array() < 0)
129  .any()) {
130  // Something went wrong while calculating the initial values for the subcompartments.
131  // See the compute_compartment() function for details.

134  // InfectedSevere.
135  init.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
136  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>()) =
137  compute_compartment<InfectionState::InfectedSevere>(
138  (Eigen::Index)InfectionTransition::InfectedSymptomsToInfectedSevere,
139  1 / m_model.parameters.template get<TimeInfectedSevere>());

141  if ((init.segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
142  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>())

145  // Something went wrong while calculating the initial values for the subcompartments.
146  // See the compute_compartment() function for details.
147  return true;
148  }
149  // InfectedCritical.
150  init.segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
151  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>()) =
152  compute_compartment<InfectionState::InfectedCritical>(
153  (Eigen::Index)InfectionTransition::InfectedSevereToInfectedCritical,
154  1 / m_model.parameters.template get<TimeInfectedCritical>());
155 
156  if ((init.segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
157  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>())
158  .array() < 0)
159  .any()) {
160  // Something went wrong while calculating the initial values for the subcompartments.
161  // See the compute_compartment() function for details.
162  return true;
163  }
164  // Recovered.
165  // Number of recovered is equal to the cumulative number of confirmed cases minus the number of people who are infected at the moment.
166  init[LctState::Count - 2] =
167  total_confirmed_cases -
168  init.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
169  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>() +
170  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>() +
171  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>())
172  .sum() -
173  deaths;
174 
175  // Susceptibles.
176  init[0] = total_population - init.segment(1, LctState::Count - 2).sum() - deaths;
177 
178  // Dead.
179  init[LctState::Count - 1] = deaths;
180 
181  for (size_t i : {(size_t)0, LctState::Count - 2, LctState::Count - 1}) {
182  if (init[i] < 0) {
183  log_error("Initialization failed. Values of total_confirmed_cases, deaths and total_population do not "
184  "match the specified values for the transitions, leading to at least one negative result.");
185  return true;
186  }
187  }
188 
189  // Update initial value vector of the model.
190  for (size_t i = 0; i < LctState::Count; i++) {
191  m_model.populations[mio::Index<LctState>(i)] = init[i];
192  }
193 
194  // Check if constraints are fulfilled.
195  return check_constraints();
196  }

213  bool check_constraints() const
214  {
215  if (!((Eigen::Index)InfectionTransition::Count == m_flows.get_num_elements())) {
216  log_error("Initial condition size does not match subcompartments.");
217  return true;
218  }
219 
220  if (!(floating_point_equal(0., m_flows.get_last_time(), 1e-8, 1e-14))) {
221  log_error("Last time point in flows has to be 0.");
222  return true;
223  }
224 
225  for (int i = 1; i < m_flows.get_num_time_points(); i++) {
226  if (!(floating_point_equal(m_dt, m_flows.get_time(i) - m_flows.get_time(i - 1), 1e-8, 1e-14))) {
227  log_error("Time points in flows have to be equidistant.");
228  return true;
229  }
230  }
231 
232  if (!(m_dt < 1)) {
233  log_warning("Step size was set very large. The result could be distorted.");
234  return true;
235  }
236 
237  // Check if model is valid and the calculated initial value vector is valid.
238  m_model.check_constraints();
239  return false;

253  template <InfectionState State>
254  Eigen::VectorXd compute_compartment(Eigen::Index idx_incoming_flow, ScalarType transition_rate) const
255  {
256  size_t num_subcompartments = LctState::template get_num_subcompartments<State>();
257  Eigen::VectorXd subcompartments(num_subcompartments);
258  // Initialize relevant density for the InfectionState.
259  // For the first subcompartment a shape parameter of one is needed.
260  ErlangDensity erlang(1, 1. / (num_subcompartments * transition_rate));
261 
262  // Initialize other relevant parameters.
263  ScalarType calc_time{0};
264  Eigen::Index calc_time_index{0};
265  ScalarType state_age{0};
266  ScalarType sum{0};
267  Eigen::Index num_time_points = m_flows.get_num_time_points();
268 
269  // Calculate number of people in each subcompartment.
270  for (size_t j = 0; j < num_subcompartments; j++) {
271  // For subcompartment number j+1, shape parameter j+1 is needed.
272  erlang.set_distribution_parameter((ScalarType)j + 1);
273 
274  // Determine relevant calculation area and corresponding index.
275  calc_time = erlang.get_support_max(m_dt, m_tol);
276  calc_time_index = (Eigen::Index)std::ceil(calc_time / m_dt) - 1;
277 
278  if (num_time_points < calc_time_index) {
279  log_error("Initialization failed. Not enough time points for the transitions are given. More than {} "
280  "are needed but just {} are given.",
281  calc_time_index, num_time_points);
282  subcompartments[j] = -1.;
283  return subcompartments;
284  }
285  else {
286  // Approximate integral with non-standard scheme.
287  for (Eigen::Index i = num_time_points - calc_time_index; i < num_time_points; i++) {
288  state_age = (num_time_points - i) * m_dt;
289  sum += erlang.eval(state_age) * m_flows[i][idx_incoming_flow];
290  }
291  subcompartments[j] = 1. / (num_subcompartments * transition_rate) * sum;
292  if (subcompartments[j] < 0) {
293  log_error(
294  "Initialization failed. Result for at least one subcompartment is less than zero. Please check "
295  "the input values.");
296  subcompartments[j] = -1.;
297  return subcompartments;
298  }
299  }
300  sum = 0;
301  }
302  return subcompartments;
303  }
