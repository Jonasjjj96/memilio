59  void get_flows(Eigen::Ref<const Vector<ScalarType>> pop, Eigen::Ref<const Vector<ScalarType>> y, ScalarType t,
60  Eigen::Ref<Vector<ScalarType>> flows) const
61  {
62  auto& params = this->parameters;
63  params.get<ContactPatterns>().get_matrix_at(t)(0, 0);
64  ScalarType coeffStoIV1 = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
65  params.get<TransmissionProbabilityOnContactV1>() / populations.get_total();
66  ScalarType coeffStoIV2 = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
67  params.get<TransmissionProbabilityOnContactV2>() / populations.get_total();
68 
69  // Normal distributed values for the stochastic part of the flows, variables are encoded
70  // in the following way: x_y is the stochastic part for the flow from x to y. Variant
71  // specific compartments also get an addendum v1 or v2 denoting the relevant variant.
72 
73  ScalarType s_ev1 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
74  ScalarType s_ev2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
75  ScalarType ev1_iv1 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
76  ScalarType ev2_iv2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
77  ScalarType iv1_rv1 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
78  ScalarType iv2_rv2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
79  ScalarType rv1_ev1v2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
80  ScalarType ev1v2_iv1v2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
81  ScalarType iv1v2_rv1v2 = mio::DistributionAdapter<std::normal_distribution<ScalarType>>::get_instance()(rng, 0.0, 1.0);
82 
83  // Assuming that no person can change its InfectionState twice in a single time step,
84  // take the minimum of the calculated flow and the source compartment, to ensure that
85  // no compartment attains negative values.
86 
87  // Calculate inv_step_size and inv_sqrt_step_size for optimization.
88  ScalarType inv_step_size = 1.0 / step_size;
89  ScalarType inv_sqrt_step_size = 1.0 / sqrt(step_size);
90 
91  // Two outgoing flows from S so will clamp their sum to S * inv_step_size to ensure non-negative S.
92  const ScalarType outflow1 = std::clamp(
93  coeffStoIV1 * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::InfectedV1] +
94  sqrt(coeffStoIV1 * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::InfectedV1])
95  * inv_sqrt_step_size * s_ev1,
96  0.0, y[(size_t)InfectionState::Susceptible] * inv_step_size);
97 
98  const ScalarType outflow2 = std::clamp(
99  coeffStoIV1 * y[(size_t)InfectionState::Susceptible] *
100  (pop[(size_t)InfectionState::InfectedV1V2] + pop[(size_t)InfectionState::InfectedV2]) +
101  sqrt(coeffStoIV2 * y[(size_t)InfectionState::Susceptible] * (pop[(size_t)InfectionState::InfectedV1V2]
102  + pop[(size_t)InfectionState::InfectedV2])) * inv_sqrt_step_size * s_ev2,
103  0.0, y[(size_t)InfectionState::Susceptible] * inv_step_size);
104 
105  const ScalarType outflow_sum = outflow1 + outflow2;
106  if (outflow_sum > 0)
107  {
108  const ScalarType scale = std::clamp(outflow_sum, 0.0 , y[(size_t)InfectionState::Susceptible] * inv_step_size) / outflow_sum;
109  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV1>()] = outflow1 * scale;
110  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV2>()] = outflow2 * scale;
111  } else
112  {
113  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV1>()] = 0;
114  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::ExposedV2>()] = 0;
115  }
116 
117  flows[get_flat_flow_index<InfectionState::ExposedV1, InfectionState::InfectedV1>()] = std::clamp(
118  (1.0 / params.get<TimeExposedV1>()) * y[(size_t)InfectionState::ExposedV1] +
119  sqrt((1.0 / params.get<TimeExposedV1>()) * y[(size_t)InfectionState::ExposedV1]) * inv_sqrt_step_size * ev1_iv1,
120  0.0, y[(size_t)InfectionState::ExposedV1] * inv_step_size);
121 
122  flows[get_flat_flow_index<InfectionState::ExposedV2, InfectionState::InfectedV2>()] = std::clamp(
123  (1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV2] +
124  sqrt((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV2]) * inv_sqrt_step_size * ev2_iv2,
125  0.0, y[(size_t)InfectionState::ExposedV2] * inv_step_size);
126 
127  flows[get_flat_flow_index<InfectionState::InfectedV1, InfectionState::RecoveredV1>()] = std::clamp(
128  (1.0 / params.get<TimeInfectedV1>()) * y[(size_t)InfectionState::InfectedV1] +
129  sqrt((1.0 / params.get<TimeInfectedV1>()) * y[(size_t)InfectionState::InfectedV1]) * inv_sqrt_step_size * iv1_rv1,
130  0.0, y[(size_t)InfectionState::InfectedV1] * inv_step_size);
131 
132  flows[get_flat_flow_index<InfectionState::InfectedV2, InfectionState::RecoveredV2>()] = std::clamp(
133  (1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV2] +
134  sqrt((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV2]) * inv_sqrt_step_size * iv2_rv2,
135  0.0, y[(size_t)InfectionState::InfectedV2] * inv_step_size);
136 
137  flows[get_flat_flow_index<InfectionState::RecoveredV1, InfectionState::ExposedV1V2>()] = std::clamp(
138  coeffStoIV2 * y[(size_t)InfectionState::RecoveredV1]
139  * (pop[(size_t)InfectionState::InfectedV1V2]+ pop[(size_t)InfectionState::InfectedV2]) +
140  sqrt(coeffStoIV2 * y[(size_t)InfectionState::RecoveredV1] * (pop[(size_t)InfectionState::InfectedV1V2]
141  + pop[(size_t)InfectionState::InfectedV2])) * inv_sqrt_step_size * rv1_ev1v2,
142  0.0, y[(size_t)InfectionState::RecoveredV1] * inv_step_size);
143 
144  flows[get_flat_flow_index<InfectionState::ExposedV1V2, InfectionState::InfectedV1V2>()] = std::clamp(
145  (1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV1V2] +
146  sqrt((1.0 / params.get<TimeExposedV2>()) * y[(size_t)InfectionState::ExposedV1V2]) /
147  sqrt(step_size) * ev1v2_iv1v2,
148  0.0, y[(size_t)InfectionState::ExposedV1V2] * inv_step_size);
149 
150  flows[get_flat_flow_index<InfectionState::InfectedV1V2, InfectionState::RecoveredV1V2>()] = std::clamp(
151  (1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV1V2] +
152  sqrt((1.0 / params.get<TimeInfectedV2>()) * y[(size_t)InfectionState::InfectedV1V2]) /
153  sqrt(step_size) * iv1v2_rv1v2,
154  0.0, y[(size_t)InfectionState::InfectedV1V2] * inv_step_size);
155  }

161 };
