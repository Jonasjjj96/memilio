84  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, ScalarType t,
85  Eigen::Ref<Eigen::VectorXd> dydt) const override
86  {
87  dydt.setZero();
88 
89  auto params = this->parameters;
90  auto total_population = pop.sum() - pop[LctState::template get_first_index<InfectionState::Dead>()];
91 
92  ScalarType infectedNoSymptoms = 0;
93  ScalarType infectedSymptoms = 0;
94  ScalarType flow = 0;
95 
96  // Calculate sum of all subcompartments for InfectedNoSymptoms.
97  infectedNoSymptoms =
98  pop.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
99  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
100  .sum();
101  // Calculate sum of all subcompartments for InfectedSymptoms.
102  infectedSymptoms = pop.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
103  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
104  .sum();
105 
106  // S'
107  ScalarType season_val = 1 + params.template get<Seasonality>() *
108  sin(3.141592653589793 * ((params.template get<StartDay>() + t) / 182.5 + 0.5));
109  dydt[0] = -y[0] / total_population * season_val * params.template get<TransmissionProbabilityOnContact>() *
110  params.template get<ContactPatterns>().get_cont_freq_mat().get_matrix_at(t)(0, 0) *
111  (params.template get<RelativeTransmissionNoSymptoms>() * infectedNoSymptoms +
112  params.template get<RiskOfInfectionFromSymptomatic>() * infectedSymptoms);
113 
114  // E'
115  dydt[1] = -dydt[0];
116  for (size_t i = 0; i < LctState::template get_num_subcompartments<InfectionState::Exposed>(); i++) {
117  // Dummy stores the value of the flow from dydt[1 + i] to dydt[2 + i].
118  // 1+i is always the index of a (sub-)compartment of E and 2+i can also be the index of the first (sub-)compartment of C.
119  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::Exposed>() *
120  (1 / params.template get<TimeExposed>()) * y[1 + i];
121  // Subtract flow from dydt[1 + i] and add to dydt[2 + i].
122  dydt[1 + i] = dydt[1 + i] - flow;
123  dydt[2 + i] = flow;
124  }
125 
126  // C'
127  for (size_t i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>(); i++) {
128  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>() *
129  (1 / params.template get<TimeInfectedNoSymptoms>()) *
130  y[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i];
131  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i] =
132  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i] - flow;
133  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i + 1] = flow;
134  }
135 
136  // I'
137  // Flow from last (sub-) compartment of C must be split between I_1 and R.
138  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
139  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] *
140  params.template get<RecoveredPerInfectedNoSymptoms>();
141  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] =
142  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] *
143  (1 - params.template get<RecoveredPerInfectedNoSymptoms>());
144 
145  for (size_t i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>(); i++) {
146  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>() *
147  (1 / params.template get<TimeInfectedSymptoms>()) *
148  y[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i];
149  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i] =
150  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i] - flow;
151  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i + 1] = flow;
152  }
153 
154  // H'
155  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
156  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
157  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] *
158  (1 - params.template get<SeverePerInfectedSymptoms>());
159  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] =
160  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] *
161  params.template get<SeverePerInfectedSymptoms>();
162  for (size_t i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedSevere>(); i++) {
163  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::InfectedSevere>() *
164  (1 / params.template get<TimeInfectedSevere>()) *
165  y[LctState::template get_first_index<InfectionState::InfectedSevere>() + i];
166  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i] =
167  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i] - flow;
168  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i + 1] = flow;
169  }
170 
171  // U'
172  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
173  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
174  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] *
175  (1 - params.template get<CriticalPerSevere>());
176  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] =
177  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] *
178  params.template get<CriticalPerSevere>();
179  for (size_t i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() - 1;
180  i++) {
181  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() *
182  (1 / params.template get<TimeInfectedCritical>()) *
183  y[LctState::template get_first_index<InfectionState::InfectedCritical>() + i];
184  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i] =
185  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i] - flow;
186  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i + 1] = flow;
187  }
188  // Last flow from U has to be divided between R and D.
189  // Must be calculated separately in order not to overwrite the already calculated values ​​for R.
190  flow = (ScalarType)LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() *
191  (1 / params.template get<TimeInfectedCritical>()) *
192  y[LctState::template get_first_index<InfectionState::Recovered>() - 1];
193  dydt[LctState::template get_first_index<InfectionState::Recovered>() - 1] =
194  dydt[LctState::template get_first_index<InfectionState::Recovered>() - 1] - flow;
195  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
196  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
197  (1 - params.template get<DeathsPerCritical>()) * flow;
198  dydt[LctState::template get_first_index<InfectionState::Dead>()] =
199  params.template get<DeathsPerCritical>() * flow;
200  }

214  TimeSeries<ScalarType> calculate_compartments(const TimeSeries<ScalarType>& subcompartments_ts) const
215  {
216  TimeSeries<ScalarType> compartments_ts((Eigen::Index)InfectionState::Count);
217  if (!(LctState::Count == subcompartments_ts.get_num_elements())) {
218  log_error("Result does not match infectionState of the Model.");
219  Eigen::VectorXd wrong_size = Eigen::VectorXd::Constant((Eigen::Index)InfectionState::Count, -1);
220  compartments_ts.add_time_point(-1, wrong_size);
221  return compartments_ts;
222  }
223  Eigen::VectorXd compartments((Eigen::Index)InfectionState::Count);
224  for (Eigen::Index i = 0; i < subcompartments_ts.get_num_time_points(); ++i) {
225  // Use segment of vector of the result with subcompartments of InfectionState with index j and sum up values of subcompartments.
226  compartments[(Eigen::Index)InfectionState::Susceptible] = subcompartments_ts[i][0];
227  compartments[(Eigen::Index)InfectionState::Exposed] =
228  subcompartments_ts[i]
229  .segment(LctState::template get_first_index<InfectionState::Exposed>(),
230  LctState::template get_num_subcompartments<InfectionState::Exposed>())
231  .sum();
232  compartments[(Eigen::Index)InfectionState::InfectedNoSymptoms] =
233  subcompartments_ts[i]
234  .segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
235  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
236  .sum();
237  compartments[(Eigen::Index)InfectionState::InfectedSymptoms] =
238  subcompartments_ts[i]
239  .segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
240  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
241  .sum();
242  compartments[(Eigen::Index)InfectionState::InfectedSevere] =
243  subcompartments_ts[i]
244  .segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
245  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>())
246  .sum();
247  compartments[(Eigen::Index)InfectionState::InfectedCritical] =
248  subcompartments_ts[i]
249  .segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
250  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>())
251  .sum();
252  compartments[(Eigen::Index)InfectionState::Recovered] =
253  subcompartments_ts[i][LctState::template get_first_index<InfectionState::Recovered>()];
254  compartments[(Eigen::Index)InfectionState::Dead] =
255  subcompartments_ts[i][LctState::template get_first_index<InfectionState::Dead>()];
256 
257  compartments_ts.add_time_point(subcompartments_ts.get_time(i), compartments);
258  }
259 
260  return compartments_ts;
261  }
